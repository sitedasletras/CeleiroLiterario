function validarCodigo(){ let codigo = document.getElementById("codigo").value; if(codigo === "CELEIRO-10"){ localStorage.setItem("tempo",10); } else if(codigo === "CELEIRO-20"){ localStorage.setItem("tempo",20); } else if(codigo === "CELEIRO-30"){ localStorage.setItem("tempo",30); } else if(codigo === "CELEIRO-60"){ localStorage.setItem("tempo",60); } else{ alert("Código inválido"); return; } localStorage.setItem("pagamento",true); }
